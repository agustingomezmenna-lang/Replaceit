    // Map column indices to header names from header row (1-based)

    function getHeaderMap(ws, headerRow1){

      if (!ws['!ref']) return new Map();

      const range = safeDecodeRange(ws['!ref']);

      if (!range) return new Map();

      const rowIdx = Math.max(1, headerRow1) - 1;

      const map = new Map(); // nameLower -> colIndex

      for (let C=range.s.c; C<=range.e.c; C++){

        const addr = XLSX.utils.encode_cell({r: rowIdx, c: C});

        const cell = ws[addr];

        const v = cell?.v;

        if (typeof v === 'string' && v.trim()){

          map.set(v.trim().toLowerCase(), C);

        }

      }

      return map;

    }

 

    function detectFinancialColumns(ws, headerMap){

      // Heurística simple: headers comunes

      const keywords = [

        'amount','importe','monto','total','saldo','balance','price','precio','cost','costo','revenue','ingreso',

        'debit','débito','debitó','credito','crédito','tax','iva','fee','comisión','commission','rate','tasa','fx','tc','tipo de cambio'

      ];

      const cols = new Set();

      for (const [nameLower, colIndex] of headerMap.entries()){

        for (const k of keywords){

          if (nameLower.includes(k)){

            cols.add(colIndex);

            break;

          }

        }

      }

      return cols;

    }

 

    function getSelectedColumns(ws, headerMap){

      const names = parseListCSV($colNames.value).map(x=>x.toLowerCase());

      const selected = new Set();

 

      // Explicit columns by header name

      for (const n of names){

        const col = headerMap.get(n);

        if (col !== undefined) selected.add(col);

      }

 

      // Auto detect (adds)

      if ($autoCols.checked){

        const auto = detectFinancialColumns(ws, headerMap);

        for (const c of auto) selected.add(c);

      }

 

      // If neither explicit nor auto yields anything, empty set means "all columns"

      return selected;

    }

 

    function isColumnAllowed(colIndex, selectedColsSet){

    }

    function applyActionToString(s, action){

      const id = action.toolId;

 

      switch (id){

        case 'dot2comma': return s.replaceAll('.', ',');

        case 'comma2dot': return s.replaceAll(',', '.');

        case 'dollarToUSD': return s.replaceAll('$', 'USD');

        case 'trim': return s.trim();

        case 'collapseSpaces': return s.replace(/\s+/g,' ');

        case 'custom': {

          const find = action.config.find ?? '';

          const repl = action.config.replace ?? '';

          if (!find) return s;

          return s.split(find).join(repl);

        }

        case 'removeCurrencySymbols': {

          const list = action.config.remove ?? ['$','U$S','USD','ARS','AR$','US$'];

          let out = s;

          for (const token of list){

            if (!token) continue;

            out = out.split(token).join('');

          }

          return out;

        }

        case 'normalizeNumberText_AR':

          return normalizeNumberText(s, 'AR');

        case 'normalizeNumberText_US':

          return normalizeNumberText(s, 'US');

        default:

          return s;

      }

    }

 

    // Heurística: si un string se parece a número con miles/decimales, lo normaliza.

    // AR: miles "." decimal ","

    // US: miles "," decimal "."

    function normalizeNumberText(s, target){

      const original = s;

 

      // Quick exits: if it has letters, skip (avoid messing IDs like "AB.12")

      if (/[A-Za-z]/.test(s)) return s;

 

      // Strip spaces

      const t = s.trim();

 

      // Must contain at least one digit

      if (!/\d/.test(t)) return s;

 

      // Allow leading currency or parentheses? We'll keep simple and only work on "pure-ish" numbers

      // Extract possible number token from string if it contains currency symbols etc.

      // We'll normalize tokens that look like: 1,234.56 or 1.234,56 or 1234.56 or 1234,56

      // Keep minus and parentheses.

      const m = t.match(/^(\(?-?)([\d.,]+)(\)?)$/);

      if (!m) return s;

 

      const prefix = m[1] || '';

      const num = m[2] || '';

      const suffix = m[3] || '';

 

      // Determine current style

      const hasDot = num.includes('.');

      const hasComma = num.includes(',');

 

      // If both dot and comma exist, infer based on last separator as decimal

      let decimalSep = null;

      if (hasDot && hasComma){

        const lastDot = num.lastIndexOf('.');

        const lastComma = num.lastIndexOf(',');

        decimalSep = (lastDot > lastComma) ? '.' : ',';

      } else if (hasDot){

        // could be decimal or thousands; assume decimal if last group length is 2-3 (common)

        const parts = num.split('.');

        const last = parts[parts.length-1];

        decimalSep = (last.length === 2) ? '.' : null; // else treat as thousands-only

      } else if (hasComma){

        const parts = num.split(',');

        const last = parts[parts.length-1];

        decimalSep = (last.length === 2) ? ',' : null;

      }

 

      // Build raw digits and decimals

      let intPart = num;

      let decPart = '';

 

      if (decimalSep){

        const idx = num.lastIndexOf(decimalSep);

        intPart = num.slice(0, idx);

        decPart = num.slice(idx+1);

      }

 

      // Remove thousands separators from intPart (both . and ,)

      const intDigits = intPart.replaceAll('.','').replaceAll(',','');

 

      // If intDigits becomes empty, bail

      if (!intDigits) return s;

 

      // Compose target

      const thousandsSep = (target === 'AR') ? '.' : ',';

      const targetDec = (target === 'AR') ? ',' : '.';

 

      const grouped = groupThousands(intDigits, thousandsSep);

      const outNum = decPart ? `${grouped}${targetDec}${decPart}` : grouped;

 

      return `${prefix}${outNum}${suffix}`.replace(original.trim(), outNum) && (t === original.trim())

        ? `${prefix}${outNum}${suffix}`

        : `${prefix}${outNum}${suffix}`;

    }

 

    function groupThousands(digits, sep){

      // digits: only 0-9

      const s = String(digits);

      let out = '';

      let count = 0;

      for (let i = s.length - 1; i >= 0; i--){

        out = s[i] + out;

        count++;

        if (count === 3 && i !== 0){

          out = sep + out;

          count = 0;

        }

      }

      return out;

    }

 

    // ========= Core processing =========

    async function processWorkbook(arrayBuffer, opts){

      const wb = XLSX.read(arrayBuffer, {type:'array'});

      const changes = []; // {sheet, cell, before, after}

      let changedCells = 0;

 

      const targetSheets = getTargetSheets(wb);

 

      for (const sn of targetSheets){

        const ws = wb.Sheets[sn];

        if (!ws) continue;

 

        const headerRow = Math.max(1, parseInt($headerRow.value || '1', 10));

        const headerMap = getHeaderMap(ws, headerRow);

        const selectedCols = getSelectedColumns(ws, headerMap);

 

        // Determine working range

        const range = getTargetRange(ws);

        if (!range) continue;

 

        for (let R=range.s.r; R<=range.e.r; R++){

          for (let C=range.s.c; C<=range.e.c; C++){

            if (!isColumnAllowed(C, selectedCols)) continue;

 

            const addr = XLSX.utils.encode_cell({r:R,c:C});

            const cell = ws[addr];

            if (!cell) continue;

 

            const v = cell.v;

 

            // Only text by default

            if (typeof v === 'string'){

              let out = v;

              for (const action of opts.pipeline){

                out = applyActionToString(out, action);

              }

              if (out !== v){

                cell.v = out;

                cell.t = 's';

                changedCells++;

                if (changes.length < opts.maxPreview){

                  changes.push({sheet: sn, cell: addr, before: v, after: out});

                }

              }

              continue;

            }

 

            // Numbers optional (convert to text)

            if (typeof v === 'number' && opts.alsoNumbers){

              let out = String(v);

              for (const action of opts.pipeline){

                out = applyActionToString(out, action);

              }

              if (out !== String(v)){

                cell.v = out;

                cell.t = 's';

                changedCells++;

                if (changes.length < opts.maxPreview){

                  changes.push({sheet: sn, cell: addr, before: String(v), after: out});

                }

              }

              continue;

            }

          }

        }

      }

 

      const outBuf = XLSX.write(wb, {bookType:'xlsx', type:'array'});

      return { outBuf, changedCells, changes };

    }

 

    function pipelineIsValid(){

      if (!pipeline.length) return {ok:false, msg:'Agregá al menos 1 acción al pipeline.'};

      // Validate custom actions

      for (const a of pipeline){

        if (a.toolId === 'custom'){

          const f = (a.config.find ?? '').trim();

          if (!f) return {ok:false, msg:'Tenés una acción Custom sin “Buscar”. Seleccionala y completala.'};

        }

      }

      return {ok:true, msg:''};

    }

 

    function renderPreviewRows(changes){

      if (!changes.length){

        $previewBody.innerHTML = `<tr><td colspan="4" class="muted">No se detectaron cambios con el pipeline actual (en el primer archivo).</td></tr>`;

        return;

      }

      $previewBody.innerHTML = changes.map(ch => `

        <tr>

          <td>${escapeHtml(ch.sheet)}</td>

          <td>${escapeHtml(ch.cell)}</td>

          <td>${escapeHtml(ch.before)}</td>

          <td class="ok">${escapeHtml(ch.after)}</td>

        </tr>

      `).join('');

    }

    $btnPreview.addEventListener('click', async () => {

      const files = [...($files.files || [])];

      if (!files.length) return;

 

      const v = pipelineIsValid();

      if (!v.ok){ alert(v.msg); return; }

 

      setStatus('Previsualizando…');

      resetLog();

      log('Preview: usando el primer archivo');

      log(`Pipeline (${pipeline.length}): ${pipeline.map(a=>a.toolId).join(' -> ')}`);

 

      try{

        const f = files[0];

        const buf = await f.arrayBuffer();

        const res = await processWorkbook(buf, {

          pipeline,

          alsoNumbers: $alsoNumbers.checked,

          maxPreview: 60

        });

 

        log(`Cambios detectados (primer archivo): ${res.changedCells}`);

        renderPreviewRows(res.changes);

        setStatus('Preview listo ✅');

      } catch(e){

        console.error(e);

        setStatus('Error ❌');

        alert('Error en preview: ' + (e?.message ?? String(e)));

      }

    });

 

    // ========= Run button =========

    $btnRun.addEventListener('click', async () => {

      const files = [...($files.files || [])];

      if (!files.length) return;

 

      const v = pipelineIsValid();

      if (!v.ok){ alert(v.msg); return; }

 

      // Default safe behavior: only text checked should be ON; we respect it anyway.

      // (Note: we already only touch strings unless alsoNumbers is checked.)

      const zipMode = $zipAll.checked && files.length > 1;

 

      setStatus('Procesando…');

      resetLog();

      log(`Archivos: ${files.length}`);

      log(`Pipeline (${pipeline.length}): ${pipeline.map(a=>a.toolId).join(' -> ')}`);

      log(`Modo: ${$alsoNumbers.checked ? 'Texto + números (a texto ⚠️)' : 'Solo texto (seguro)'} `);

      log(`Sheets: ${$sheets.value ? $sheets.value : '(todas)'}`);

      log(`Range: ${$rangeA1.value ? $rangeA1.value : '(todo)'}`);

      log(`Columnas header: ${$colNames.value ? $colNames.value : '(todas)'} | AutoCols: ${$autoCols.checked ? 'ON' : 'OFF'} | HeaderRow: ${$headerRow.value}`);

 

      $btnRun.disabled = true;

      $btnPreview.disabled = true;

 

      const zip = zipMode ? new JSZip() : null;

 

      try{

        for (let i=0;i<files.length;i++){

          const f = files[i];

          log(`\n[${i+1}/${files.length}] ${f.name}`);

 

          const buf = await f.arrayBuffer();

          const res = await processWorkbook(buf, {

            pipeline,

            alsoNumbers: $alsoNumbers.checked,

            maxPreview: 0

          });

 

          log(`Celdas modificadas: ${res.changedCells}`);

 

          const tag = pipeline.map(a=>a.toolId).join('+').slice(0, 80);

          const outName = f.name.replace(/\.xlsx$/i,'') + ` - ${tag}.xlsx`;

 

          if (zipMode){

            zip.file(outName, res.outBuf);

          } else {

            const blob = new Blob([res.outBuf], {type:'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet'});

            saveAs(blob, outName);

          }

        }

 

        if (zipMode){

          log('\nCreando ZIP…');

          const zipBlob = await zip.generateAsync({type:'blob'});

          saveAs(zipBlob, `excels_processed.zip`);

          log('ZIP descargado ✅');

        }

 

        setStatus('Listo ✅');

        log('\nFinalizado ✅');

      } catch(e){

        console.error(e);

        setStatus('Error ❌');

        alert('Error: ' + (e?.message ?? String(e)));

      } finally{

        $btnRun.disabled = false;

        $btnPreview.disabled = false;

      }

    });
